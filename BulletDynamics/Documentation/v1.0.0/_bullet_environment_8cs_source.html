144 public virtual Bullet AddBullet(BulletData bulletData, Transform outputPoint)
145 {
146 if (bulletData == null)

154 return null;
155 }
156
157 Vector3 initialPosition = outputPoint.position;
158 Vector3 initialVelocity = outputPoint.forward * bulletData.PhysicsData.InitialVelocity;
159 Bullet bullet = new Bullet(bulletData, initialPosition, initialVelocity);
