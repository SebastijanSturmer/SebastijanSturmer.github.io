109 public static float GetBarrelExitPointZeroedRotation(float distance, float bulletInitialVelocity, float dragCoefficient, float referenceArea, Vector3 gravity, float airDensity, float timeStep)
110 {
111 float distanceTolerance = (distance / 100f) * 0.0025f; // Tolerance for distance accuracy
112 float heightTolerance = (distance / 100f) * 0.0005f; // Tolerance for height accuracy
113 const int maxIterations = 10000; // Maximum number of iterations for convergence
114 const float maxAngle = 45f; // Maximum angle if target is unreachable
115 Vector3 noWind = Vector3.zero;
116 // Method to simulate the flight and find the distance using Runge-Kutta
117 Vector3 SimulateBulletFlight(float angle, out float flightTime)
118 {
119 float radianAngle = angle * Mathf.Deg2Rad;
120 Vector3 initialVelocity = Quaternion.Euler(-angle, 0, 0) * Vector3.forward * bulletInitialVelocity;
121 Vector3 position = Vector3.zero;
122 Vector3 velocity = initialVelocity;
123 flightTime = 0f;
124
125 while (position.z < distance && flightTime < maxIterations * timeStep && position.y >= 0)
126 {
127 CalculateNewBulletPositionAndVelocity(velocity, position, dragCoefficient, referenceArea, gravity, noWind, airDensity, timeStep, out Vector3 newPosition, out Vector3 newVelocity);
128
129 velocity = newVelocity;
130 position = newPosition;
131
132 flightTime += timeStep;
133 }
134
135 return position;
136 }
137
138 // Iteratively find the correct angle
139 float minAngle = 0f;
140 float optimalAngle = 0;
141 float maxAngleReachable = maxAngle;
142
143 for (int i = 0; i < maxIterations; i++)
144 {
145 float angle = (minAngle + maxAngleReachable) / 2f;
146 Vector3 simulatedEndPosition = SimulateBulletFlight(angle, out _);
147 float simulatedDistance = simulatedEndPosition.z;
148
149 if ((simulatedDistance > distance && simulatedDistance - distance < distanceTolerance && Mathf.Abs(simulatedEndPosition.y) < heightTolerance) || Mathf.Approximately(minAngle, maxAngleReachable))
150 {
151 optimalAngle = angle;
152 break;
153 }
154
155 if (simulatedDistance < distance)
156 {
157 minAngle = angle;
158 }
159 else
160 {
161 maxAngleReachable = angle;
162 }
163 }
164
165 return optimalAngle != 0 ? optimalAngle : minAngle != 0 ? (minAngle + maxAngleReachable) / 2f : maxAngleReachable;
166 }
