46 public virtual void UpdateBullets(float timeStep)
47 {
48 // Initialize values for the job.
49 Vector3 wind = Vector3.zero;
50 float maximumDistanceFromWorldOrigin = 1000f;
51 Vector3 gravity = Physics.gravity;
52 float airDensity = 0;
53 LayerMask bulletCollisionLayerMask = LayerMask.GetMask("Default");
54 if (BulletEnvironment.Instance != null)
55 {
56 gravity = BulletEnvironment.Instance.EnvironmentData.OverrideUnityGravity ? Physics.gravity : new Vector3(0, BulletEnvironment.Instance.EnvironmentData.Gravity, 0);
57 wind = BulletEnvironment.Instance.EnvironmentData.Wind;
58 maximumDistanceFromWorldOrigin = BulletEnvironment.Instance.MaximumAllowedOffsetFromWorldOrigin;
59 airDensity = BulletEnvironment.Instance.AirDensity;
60 bulletCollisionLayerMask = BulletEnvironment.Instance.BulletCollisionLayerMask;
61 }
62
63 var positions = new NativeArray<Vector3>(_bullets.Count, Allocator.TempJob);
64 var velocities = new NativeArray<Vector3>(_bullets.Count, Allocator.TempJob);
65 var bulletPhysicsData = new NativeArray<BulletData.PhysicData>(_bullets.Count, Allocator.TempJob);
66 var collisionMaterials = new NativeArray<SurfacePhysicMaterial.MaterialData>(_bullets.Count, Allocator.TempJob);
67 var expiredBullets = new NativeArray<bool>(_bullets.Count, Allocator.TempJob);
68
69 var raycastCommands = new NativeArray<RaycastCommand>(_bullets.Count, Allocator.TempJob);
70 var raycastResults = new NativeArray<RaycastHit>(_bullets.Count, Allocator.TempJob);
71 var bulletHitInfos = new NativeArray<BulletHitInfo>(_bullets.Count, Allocator.TempJob);
72
73 List<DynamicSurface> surfaces = new List<DynamicSurface>();
74
75 for (int i = 0; i < _bullets.Count; i++)
76 {
77 positions[i] = _bullets[i].Position;
78 velocities[i] = _bullets[i].Velocity;
79 bulletPhysicsData[i] = _bullets[i].Data.PhysicsData;
80 collisionMaterials[i] = default;
81 expiredBullets[i] = default;
82 raycastCommands[i] = new RaycastCommand(_bullets[i].Position, _bullets[i].Velocity.normalized, new QueryParameters(bulletCollisionLayerMask), _bullets[i].Velocity.magnitude * Time.deltaTime);
83 bulletHitInfos[i] = new BulletHitInfo();
84 }
85
86 // Process collisions.
87 JobHandle raycastJobHandle = RaycastCommand.ScheduleBatch(raycastCommands, raycastResults, 1);
88 raycastJobHandle.Complete();
89 GetAdditionalInformationFromRaycastResults(raycastResults, ref bulletHitInfos, ref collisionMaterials, ref surfaces, bulletCollisionLayerMask);
90
91 // Complete a bullets job.
92 var bulletsJob = CreateBulletJob(positions, velocities, collisionMaterials, bulletPhysicsData, bulletHitInfos, expiredBullets, wind, airDensity, gravity, timeStep);
93 JobHandle bulletsJobHandle = bulletsJob.Schedule(_bullets.Count, 64, raycastJobHandle);
94 bulletsJobHandle.Complete();
95
96 // Finalize processing by processing the bullets at this frame determining their status and if they collided or expired.
97 ProcessBulletsAtThisFrame(positions, velocities, bulletHitInfos, expiredBullets, surfaces, maximumDistanceFromWorldOrigin);
98
99 // Cleanup.
100 positions.Dispose();
101 velocities.Dispose();
102 bulletPhysicsData.Dispose();
103 collisionMaterials.Dispose();
104 bulletHitInfos.Dispose();
105 expiredBullets.Dispose();
106 raycastCommands.Dispose();
107 raycastResults.Dispose();
108 }

160 protected virtual void GetAdditionalInformationFromRaycastResults(NativeArray<RaycastHit> raycastResults, ref NativeArray<BulletHitInfo> bulletHitInfos, ref NativeArray<SurfacePhysicMaterial.MaterialData> collisionMaterials, ref List<DynamicSurface> surfaces, LayerMask bulletCollisionLayerMask)
161 {
162 for (int i = 0; i < raycastResults.Length; i++)
163 {
164 if (raycastResults[i].collider == null)
165 {
166 bulletHitInfos[i] = new BulletHitInfo();
167 continue;
168 }
169
170 var info = raycastResults[i];
171
172 // Get exit point.
173 Vector3 exitPoint = Vector3.zero;
174 Vector3 originPos = _bullets[i].Position;
175 Vector3 direction = _bullets[i].Velocity.normalized;
176
177 float maxDistance = _bullets[i].Velocity.magnitude * Time.deltaTime + RAYCAST_DISTANCE_FOR_EXIT_POINT;
178 RaycastHit[] hitResults = Physics.RaycastAll(originPos, direction, maxDistance, bulletCollisionLayerMask);
179 Array.Sort(hitResults, 0, hitResults.Length, _raycastDistanceComparer);
180
181 for (int j = 0; j < hitResults.Length; j++)
182 {
183 if (hitResults[j].collider == info.collider)
184 {
185 if (hitResults.Length == 1)
186 {
187 if (Physics.Raycast(originPos + direction * maxDistance, -direction, out RaycastHit backHitSingleTarget, maxDistance, bulletCollisionLayerMask))
188 {
189 exitPoint = backHitSingleTarget.point;
190 }
191 }
192
193 continue;
194 }
195
196 if (Physics.Raycast(hitResults[j].point, -direction, out RaycastHit backHit, Mathf.Infinity, bulletCollisionLayerMask))
197 {
198 if (j - 1 >= 0)
199 {
200 if (backHit.collider != hitResults[j - 1].collider)
201 {
202 if (Physics.Raycast(originPos + direction * maxDistance, -direction, out RaycastHit backHitSingleTarget, maxDistance, bulletCollisionLayerMask))
203 {
204 exitPoint = backHitSingleTarget.point;
205 break;
206 }
207 }
208 }
209
210 exitPoint = backHit.point;
211 break;
212 }
213 }
214
215 Vector3 force = _bullets[i].Velocity * _bullets[i].Data.PhysicsData.Mass;
216 bulletHitInfos[i] = new BulletHitInfo(true, info.point, info.normal, exitPoint, force);
217
218 // Get additional information about the collision.
219 if (raycastResults[i].collider.TryGetComponent(out DynamicSurface surface))
220 {
221 if (surface.HasData)
222 {
223 collisionMaterials[i] = surface.Data;
224 surfaces.Add(surface);
225 }
226 else
227 {
228 collisionMaterials[i] = _defaultCollisionData;
229 surfaces.Add(null);
230 }
231 }
232 else
233 {
234 collisionMaterials[i] = _defaultCollisionData;
235 surfaces.Add(null);
236 }
237 }
238 }

290 protected virtual void ProcessBulletsAtThisFrame(NativeArray<Vector3> positions, NativeArray<Vector3> velocities, NativeArray<BulletHitInfo> bulletHitInfos, NativeArray<bool> expiredBullets, List<DynamicSurface> surfaces, float maximumDistanceFromWorldOrigin)
291 {
292 List<Bullet> expiredBulletsList = new List<Bullet>();
293 int collisionIndex = 0;
294 for (int i = 0; i < _bullets.Count; i++)
295 {
296 if (bulletHitInfos[i].DidHit)
297 {
298 HandleCollision(_bullets[i], positions[i], bulletHitInfos[i], surfaces[collisionIndex]);
299 bulletHitInfos[i] = new BulletHitInfo();
300 collisionIndex++;
301 }
302 if (expiredBullets[i] || Vector3.Distance(positions[i], Vector3.zero) > maximumDistanceFromWorldOrigin)
303 {
304 expiredBulletsList.Add(_bullets[i]);
305 }
306
307 _bullets[i].Position = positions[i];
308 _bullets[i].Velocity = velocities[i];
309 }
310
311 foreach (Bullet expiredBullet in expiredBulletsList)
312 {
313 HandleExpiration(expiredBullet);
314 }
315 }
