135 protected virtual GameObject CreateBulletVisual(Bullet bullet)
136 {
137 if (bullet == null || bullet.Data == null || bullet.Data.Prefab == null)
138 {
139 return null;
140 }
141
142 if (!_objectPoolsForBullets.ContainsKey(bullet.Data))
143 {
144 _objectPoolsForBullets.Add(bullet.Data, new ObjectPool<GameObject>(bullet.Data.Prefab));
145 }
146 ObjectPool<GameObject> objectPool = _objectPoolsForBullets[bullet.Data];
147 GameObject bulletGO = objectPool.GetObject(bullet.Position, bullet.Velocity.normalized);
148 bulletGO.transform.parent = _bulletHolder;

161 protected virtual GameObject CreateHitEffect(DynamicSurface surface, BulletHitInfo hitInfo, Bullet bullet)
162 {
163 GameObject hitEffectGameObject = null;
164 SurfaceOnHitEffect hitEffect = null;
165 SurfacePhysicMaterial material = null;
166 if (surface != null && surface.HasData && surface.SurfacePhysicMaterial.OnHitEffect != null)
167 {
168 if (surface.SurfacePhysicMaterial.OnHitEffect.GetComponent<SurfaceOnHitEffect>() == null)
169 {
170 Debug.LogError($"{nameof(BulletVisualsHandler)} :: {nameof(CreateHitEffect)} :: Couldn't create hit effect for surface {surface.gameObject.name} because it doesn't have component {nameof(SurfaceOnHitEffect)} assigned.");
171 }
172 else
173 {
174 if (!_objectPoolsForSurfaceHits.ContainsKey(surface.SurfacePhysicMaterial))
175 {
176 _objectPoolsForSurfaceHits.Add(surface.SurfacePhysicMaterial, new ObjectPool<SurfaceOnHitEffect>(surface.SurfacePhysicMaterial.OnHitEffect.GetComponent<SurfaceOnHitEffect>()));
177 }
178 material = surface.SurfacePhysicMaterial;
179 ObjectPool<SurfaceOnHitEffect> objectPool = _objectPoolsForSurfaceHits[material];
180 hitEffect = objectPool.GetObject(hitInfo.EnterPoint, hitInfo.Normal);
181 }
182 }
183 else
184 {
185 hitEffect = _objectPoolForDefaultHitEffect.GetObject(hitInfo.EnterPoint, hitInfo.Normal);
186 }
187
188 if (hitEffect != null)
189 {
190 hitEffect.Setup(material);
191 hitEffect.OnCompleted.AddListener(OnSurfaceHitCompleted);
192 hitEffectGameObject = hitEffect.gameObject;
193 hitEffectGameObject.transform.parent = _bulletOnHitEffectsHolder;
194
195 hitEffect.StartEffect();
196 }
197 return hitEffectGameObject;
198 }
