121 private void HandleWeapon()
122 {
123 if (Input.GetMouseButton(0))
124 {
125 _weapon.Shoot();
126 }
127
128 _isZoomed = Input.GetMouseButton(1);
129 if (_isZoomed)
130 {
131 _weapon.transform.localPosition = _scopedWeaponPosition;
132 _camera.fieldOfView = _scopedCameraFOV;
133 }
134 else
135 {
136 _weapon.transform.localPosition = _initialWeaponPosition;
137 _camera.fieldOfView = _defaultCameraFOV;
138 }
139 _crosshair.SetActive(!_isZoomed);
140
141 if (Input.GetKeyDown(KeyCode.R))
142 {
143 _weaponMagazine.Reload(_weaponReloadDuration);
144 }
145
146 RaycastHit hit;
147 if (Physics.Raycast(_camera.transform.position, _camera.transform.forward, out hit, Mathf.Infinity, BulletEnvironment.Instance.BulletCollisionLayerMask))
148 {
149 _weapon.OutputPoint.parent.LookAt(hit.point);
150 if (Vector3.Angle(_weapon.OutputPoint.forward, _weapon.transform.forward) > MAX_OUTPUT_POINT_ANGLE)
151 {
152 _weapon.OutputPoint.parent.localRotation = Quaternion.identity;
153 }
154 }
155 }

160 private void HandleMovement()

186 private void HandleCamera()
187 {
188 float sensitivity = _isZoomed ? _mouseSensitivity * ZOOMED_SENSITIVITY_MULTIPLIER : _mouseSensitivity;
189 float xRotation = Input.GetAxis("Mouse X") * sensitivity;
190 float yRotation = -Input.GetAxis("Mouse Y") * sensitivity;
191
192 _camera.transform.localRotation = Quaternion.Euler(_camera.transform.localEulerAngles.x + yRotation,0,0);
193 _currentRotation = _currentRotation * Quaternion.Euler(0, xRotation, 0);
194 transform.rotation = _currentRotation;

211 }
