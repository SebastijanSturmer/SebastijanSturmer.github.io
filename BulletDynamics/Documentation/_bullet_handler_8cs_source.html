40 public virtual void UpdateBullets(float timeStep)
41 {
42 // Initialize values for the job.
43 Vector3 wind = Vector3.zero;
44 float maximumDistanceFromWorldOrigin = 1000f;
45 Vector3 gravity = Physics.gravity;
46 float airDensity = 0;
47 LayerMask bulletCollisionLayerMask = LayerMask.GetMask("Default");
48 if (BulletEnvironment.Instance != null)
49 {
50 gravity = BulletEnvironment.Instance.EnvironmentData.OverrideUnityGravity ? Physics.gravity : new Vector3(0, BulletEnvironment.Instance.EnvironmentData.Gravity, 0);
51 wind = BulletEnvironment.Instance.EnvironmentData.Wind;
52 maximumDistanceFromWorldOrigin = BulletEnvironment.Instance.MaximumAllowedOffsetFromWorldOrigin;
53 airDensity = BulletEnvironment.Instance.AirDensity;
54 bulletCollisionLayerMask = BulletEnvironment.Instance.BulletCollisionLayerMask;
55 }
56
57 var positions = new NativeArray<Vector3>(_bullets.Count, Allocator.TempJob);
58 var velocities = new NativeArray<Vector3>(_bullets.Count, Allocator.TempJob);
59 var bulletPhysicsData = new NativeArray<BulletData.PhysicData>(_bullets.Count, Allocator.TempJob);
60 var collisionMaterials = new NativeArray<SurfacePhysicMaterial.MaterialData>(_bullets.Count, Allocator.TempJob);
61 var expiredBullets = new NativeArray<bool>(_bullets.Count, Allocator.TempJob);
62
63 var raycastCommands = new NativeArray<RaycastCommand>(_bullets.Count, Allocator.TempJob);
64 var raycastResults = new NativeArray<RaycastHit>(_bullets.Count, Allocator.TempJob);
65 var bulletHitInfos = new NativeArray<BulletHitInfo>(_bullets.Count, Allocator.TempJob);
66
67 List<DynamicSurface> surfaces = new List<DynamicSurface>();
68
69 for (int i = 0; i < _bullets.Count; i++)
70 {
71 positions[i] = _bullets[i].Position;
72 velocities[i] = _bullets[i].Velocity;
73 bulletPhysicsData[i] = _bullets[i].Data.PhysicsData;
74 collisionMaterials[i] = default;
75 expiredBullets[i] = default;
76 raycastCommands[i] = new RaycastCommand(_bullets[i].Position, _bullets[i].Velocity.normalized, new QueryParameters(bulletCollisionLayerMask), _bullets[i].Velocity.magnitude * Time.deltaTime);
77 bulletHitInfos[i] = new BulletHitInfo();
78 }
79
80 // Process collisions.
81 JobHandle raycastJobHandle = RaycastCommand.ScheduleBatch(raycastCommands, raycastResults, 1);
82 raycastJobHandle.Complete();
83 GetAdditionalInformationFromRaycastResults(raycastResults, ref bulletHitInfos, ref collisionMaterials, ref surfaces, bulletCollisionLayerMask);
84
85 // Complete a bullets job.
86 var bulletsJob = CreateBulletJob(positions, velocities, collisionMaterials, bulletPhysicsData, bulletHitInfos, expiredBullets, wind, airDensity, gravity, timeStep);
87 JobHandle bulletsJobHandle = bulletsJob.Schedule(_bullets.Count, 64, raycastJobHandle);
88 bulletsJobHandle.Complete();
89
90 // Finalize processing by processing the bullets at this frame determining their status and if they collided or expired.
91 ProcessBulletsAtThisFrame(positions, velocities, bulletHitInfos, expiredBullets, surfaces, maximumDistanceFromWorldOrigin);
92
93 // Cleanup.
94 positions.Dispose();
95 velocities.Dispose();
96 bulletPhysicsData.Dispose();
97 collisionMaterials.Dispose();
98 bulletHitInfos.Dispose();
99 expiredBullets.Dispose();
100 raycastCommands.Dispose();
101 raycastResults.Dispose();
102 }

154 protected virtual void GetAdditionalInformationFromRaycastResults(NativeArray<RaycastHit> raycastResults, ref NativeArray<BulletHitInfo> bulletHitInfos, ref NativeArray<SurfacePhysicMaterial.MaterialData> collisionMaterials, ref List<DynamicSurface> surfaces, LayerMask bulletCollisionLayerMask)
155 {
156 for (int i = 0; i < raycastResults.Length; i++)
157 {
158 if (raycastResults[i].collider == null)
159 {
160 bulletHitInfos[i] = new BulletHitInfo();
161 continue;
162 }
163
164 var info = raycastResults[i];
165
166 // Get exit point.
167 Vector3 exitPoint = Vector3.zero;
168 Vector3 originPos = _bullets[i].Position;
169 Vector3 direction = _bullets[i].Velocity.normalized;
170
171 float maxDistance = _bullets[i].Velocity.magnitude * Time.deltaTime + RAYCAST_DISTANCE_FOR_EXIT_POINT;
172 RaycastHit[] hitResults = Physics.RaycastAll(originPos, direction, maxDistance, bulletCollisionLayerMask);
173 Array.Sort(hitResults, 0, hitResults.Length, _raycastDistanceComparer);
174
175 for (int j = 0; j < hitResults.Length; j++)
176 {
177 if (hitResults[j].collider == info.collider)
178 {
179 if (hitResults.Length == 1)
180 {
181 if (Physics.Raycast(originPos + direction * maxDistance, -direction, out RaycastHit backHitSingleTarget, maxDistance, bulletCollisionLayerMask))
182 {
183 exitPoint = backHitSingleTarget.point;
184 }
185 }
186
187 continue;
188 }
189
190 if (Physics.Raycast(hitResults[j].point, -direction, out RaycastHit backHit, Mathf.Infinity, bulletCollisionLayerMask))
191 {
192 if (j - 1 >= 0)
193 {
194 if (backHit.collider != hitResults[j - 1].collider)
195 {
196 if (Physics.Raycast(originPos + direction * maxDistance, -direction, out RaycastHit backHitSingleTarget, maxDistance, bulletCollisionLayerMask))
197 {
198 exitPoint = backHitSingleTarget.point;
199 break;
200 }
201 }
202 }
203
204 exitPoint = backHit.point;
205 break;
206 }
207 }
208
209 bulletHitInfos[i] = new BulletHitInfo(true, info.point, info.normal, exitPoint);
210
211 // Get additional information about the collision.
212 if (raycastResults[i].collider.TryGetComponent(out DynamicSurface surface))
213 {
214 if (surface.HasData)
215 {
216 collisionMaterials[i] = surface.Data;
217 surfaces.Add(surface);
218 }
219 else
220 {
221 collisionMaterials[i] = _defaultCollisionData;
222 surfaces.Add(null);
223 }
224 }
225 else
226 {
227 collisionMaterials[i] = _defaultCollisionData;
228 surfaces.Add(null);
229 }
230 }
231 }

283 protected virtual void ProcessBulletsAtThisFrame(NativeArray<Vector3> positions, NativeArray<Vector3> velocities, NativeArray<BulletHitInfo> bulletHitInfos, NativeArray<bool> expiredBullets, List<DynamicSurface> surfaces, float maximumDistanceFromWorldOrigin)
284 {
285 List<Bullet> expiredBulletsList = new List<Bullet>();
286 int collisionIndex = 0;
287 for (int i = 0; i < _bullets.Count; i++)
288 {
289 if (bulletHitInfos[i].DidHit)
290 {
291 HandleCollision(_bullets[i], positions[i], bulletHitInfos[i], surfaces[collisionIndex]);
292 bulletHitInfos[i] = new BulletHitInfo();
293 collisionIndex++;
294 }
295 if (expiredBullets[i] || Vector3.Distance(positions[i], Vector3.zero) > maximumDistanceFromWorldOrigin)
296 {
297 expiredBulletsList.Add(_bullets[i]);
298 }
299
300 _bullets[i].Position = positions[i];
301 _bullets[i].Velocity = velocities[i];
302 }
303
304 foreach (Bullet expiredBullet in expiredBulletsList)
305 {
306 HandleExpiration(expiredBullet);
307 }
308 }
